#+begin_src rustic
  fn reverse_helper<'a>(s: &'a str, acc: Vec<&'a str>) -> Vec<&'a str> {
      if s.is_empty() {
          return acc
      } else {
          reverse_helper(
              &s[1..],
              Vec::from_iter(acc.into_iter().chain(std::iter::once(&s[0..1])))
          )
      }
  }

  pub fn reverse(input: &str) -> String {
      reverse_helper(input, vec![])
          .join("")
  }
#+end_src
